/** Рекурсивный поиск факториала */
var factorial=(function(){
    var stack =[1,1,2];
    return function (n){
        if(stack[n])return stack[n];
        stack[n]=n*factorial(n-1);
        return stack[n];
    }
})();

/** Поиск полным перебором факториона */
function factorion(){
    var str,sum;
    for(var len=7*factorial(9),i= 1;i<=len;i++){
        str=i.toString();
        sum=0;
        for(var secLen=str.length,j=0; j < secLen; j++){
            sum+=factorial(parseInt(str[j]));
        }
        if(sum==i) console.log(str);
    }
}
factorion();
